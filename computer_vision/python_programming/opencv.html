矩：Moments

图像矩可以计算图像的质心，面积等等。

# 根据图像的矩计算重心 
def find_center(contour):
    M = cv2.moments(contour)
    rx,ry,rw,rh = cv2.boundingRect(contour)
    try:
        cx = int(M['m10'] / M['m00'])
        cy = int(M['m01'] / M['m00'])
        print "ROI centroid=", (cx, cy);
    except ZeroDivisionError:
        cx = rx + int(rw / 2)
        cy = ry + int(rh / 2)
        print "ROI centroid=Unknown, use b-box center=", (cx, cy)
    return cx, cy
